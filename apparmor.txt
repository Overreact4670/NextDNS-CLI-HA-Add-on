#include <tunables/global>

profile nextdns flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Allow nextdns binary to execute
  /usr/sbin/nextdns ix,
  /usr/bin/nextdns ix,

  # Allow the run script and shell
  /run.sh r,
  /bin/bash ix,
  /bin/sh ix,
  /usr/bin/with-contenv ix,

  # bashio and s6 helpers
  /usr/lib/bashio/** r,
  /usr/bin/bashio ix,
  /run/s6/** r,

  # Read add-on options from Supervisor
  /data/options.json r,

  # nextdns runtime state (cache, config)
  /data/ rw,
  /data/** rw,
  /tmp/ rw,
  /tmp/** rw,
  /run/ rw,
  /run/** rw,

  # DNS resolution cache and hosts file
  /etc/hosts r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/ssl/certs/** r,
  /etc/ca-certificates/** r,
  /usr/share/ca-certificates/** r,

  # System libraries and binaries (read only)
  /lib/** r,
  /lib64/** r,
  /usr/lib/** r,
  /usr/share/** r,

  # Process info (needed by nextdns for client identification)
  @{PROC}/@{pid}/net/dev r,
  @{PROC}/@{pid}/net/if_inet6 r,
  @{PROC}/sys/net/** r,
  @{PROC}/sys/kernel/hostname r,

  # Network access
  # nextdns needs raw network access for DNS listening and DoH upstream
  network inet dgram,
  network inet stream,
  network inet6 dgram,
  network inet6 stream,
  network netlink raw,

  # Signals
  signal (send, receive) set=(term, kill, hup, usr1, usr2),

  # Deny everything else explicitly
  deny @{PROC}/sys/kernel/core_pattern w,
  deny /sys/firmware/** rw,
  deny /sys/kernel/security/** rw,
}
