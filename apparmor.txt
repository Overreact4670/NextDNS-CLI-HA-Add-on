#include <tunables/global>

profile ADDON_SLUG flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Add-on data and options
  /data/** rw,

  # nextdns binary
  /usr/sbin/nextdns cx -> nextdns,
  /usr/bin/nextdns cx -> nextdns,

  profile nextdns flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_ADDON_SLUG,

    # nextdns binary
    /usr/sbin/nextdns r,
    /usr/bin/nextdns r,

    # Add-on data (cache, config, state)
    /data/** rw,
    /run/{,**} rwk,
    /tmp/** rwk,

    # System config
    /etc/hosts r,
    /etc/resolv.conf r,
    /etc/nsswitch.conf r,
    /etc/ssl/** r,
    /etc/ca-certificates/** r,

    # Libraries
    /lib/** rm,
    /lib64/** rm,
    /usr/lib/** rm,
    /usr/share/** r,

    # /proc for network interface and client identification
    @{PROC}/ r,
    @{PROC}/@{pid}/** r,
    @{PROC}/sys/net/** r,
    @{PROC}/sys/kernel/hostname r,

    # Network (DNS listen + DoH upstream)
    network inet dgram,
    network inet stream,
    network inet6 dgram,
    network inet6 stream,
    network netlink raw,

    # Capabilities needed for port 53 and raw sockets
    capability net_bind_service,
    capability net_admin,
    capability net_raw,

    # Deny sensitive paths
    deny @{PROC}/sys/kernel/core_pattern w,
    deny /sys/firmware/** rw,
    deny /sys/kernel/security/** rw,
  }
}
