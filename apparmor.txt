#include <tunables/global>

profile nextdns flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # -------------------------------------------------------------------------
  # S6-Overlay v3 init system (required by HA base images)
  # -------------------------------------------------------------------------
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/fix-attrs.d/** r,

  # -------------------------------------------------------------------------
  # bashio and run script
  # -------------------------------------------------------------------------
  /run.sh rx,
  /usr/lib/bashio/** r,
  /tmp/** rw,

  # -------------------------------------------------------------------------
  # nextdns binary
  # -------------------------------------------------------------------------
  /usr/sbin/nextdns ix,
  /usr/bin/nextdns ix,

  # -------------------------------------------------------------------------
  # Add-on data (config, cache, runtime state)
  # -------------------------------------------------------------------------
  /data/ rw,
  /data/** rw,
  /run/ rw,
  /run/** rw,
  /tmp/ rw,

  # -------------------------------------------------------------------------
  # System files needed at runtime
  # -------------------------------------------------------------------------
  /etc/hosts r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/ssl/certs/** r,
  /etc/ca-certificates/** r,
  /usr/share/ca-certificates/** r,

  # System libraries (read only)
  /lib/** rm,
  /lib64/** rm,
  /usr/lib/** rm,
  /usr/share/** r,

  # -------------------------------------------------------------------------
  # /proc entries for client identification
  # -------------------------------------------------------------------------
  @{PROC}/@{pid}/net/dev r,
  @{PROC}/@{pid}/net/if_inet6 r,
  @{PROC}/sys/net/** r,
  @{PROC}/sys/kernel/hostname r,

  # -------------------------------------------------------------------------
  # Network access
  # -------------------------------------------------------------------------
  network inet dgram,
  network inet stream,
  network inet6 dgram,
  network inet6 stream,
  network netlink raw,

  # -------------------------------------------------------------------------
  # Signals
  # -------------------------------------------------------------------------
  signal (send, receive) set=(term, kill, hup, usr1, usr2),

  # -------------------------------------------------------------------------
  # Explicit denies
  # -------------------------------------------------------------------------
  deny @{PROC}/sys/kernel/core_pattern w,
  deny /sys/firmware/** rw,
  deny /sys/kernel/security/** rw,
}
