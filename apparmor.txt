#include <tunables/global>

profile ADDON_SLUG flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Allow all capabilities and file access as a base
  # (required for s6-overlay and bashio to function)
  capability,
  file,

  # Specific capabilities needed by nextdns
  capability net_bind_service,
  capability net_admin,
  capability net_raw,
  capability setgid,
  capability setuid,
  capability dac_override,

  # S6-Overlay init system
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/lib/bashio/** ix,
  /etc/s6/** rix,
  /run/s6/** rix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /etc/fix-attrs.d/** r,
  /package/** ix,
  /command/** ix,

  # Run script
  /run.sh rx,
  /tmp/** rw,

  # nextdns binary
  /usr/sbin/nextdns ix,
  /usr/bin/nextdns ix,

  # Add-on data (config, cache, state)
  /data/ rw,
  /data/** rw,
  /run/ rw,
  /run/** rw,

  # System config files
  /etc/hosts r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/ssl/** r,
  /etc/ca-certificates/** r,
  /usr/share/ca-certificates/** r,

  # System libraries
  /lib/** rm,
  /lib64/** rm,
  /usr/lib/** rm,
  /usr/share/** r,

  # /proc entries for network interface and client identification
  @{PROC}/ r,
  @{PROC}/@{pid}/** r,
  @{PROC}/sys/net/** r,
  @{PROC}/sys/kernel/hostname r,

  # Network access (DNS + DoH upstream)
  network inet dgram,
  network inet stream,
  network inet6 dgram,
  network inet6 stream,
  network netlink raw,

  # Signals
  signal (send, receive) set=(term, kill, hup, usr1, usr2),

  # Deny sensitive paths
  deny @{PROC}/sys/kernel/core_pattern w,
  deny /sys/firmware/** rw,
  deny /sys/kernel/security/** rw,
}
